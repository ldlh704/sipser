    \begin{itemize}
      \item Base case:
      \begin{flalign*}
        S(1) &= \frac{1}{2}(1)(2) = 1 &&
      \end{flalign*}
      \item Inductive case:
      \begin{flalign*}
        S(n) &= S(n-1) + n \\
                &= \frac{1}{2}(n-1)(n - 1 + 1) + n \\
                &= \frac{1}{2}(n^2 + n) \\
                & = \frac{1}{2}n(n+1) &&
      \end{flalign*} 
    \end{itemize}
